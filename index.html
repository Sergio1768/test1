import requests def colocar_pixel(x, y, color): url = "https://api.rplace.tk/api/v1/place" data = { "x": x, "y": y, "color": color } response = requests.post(url, json=data) if response.status_code == 200: print("Píxel colocado exitosamente en las coordenadas ({}, {})".format(x, y)) else: print("Error al colocar el píxel:", response.text) # Colocar un píxel rojo en las coordenadas (100, 100) colocar_pixel(100, 100, "#FF0000")
